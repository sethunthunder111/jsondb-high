example.ts
import JSONDatabase from 'jsondb-high';

// Initialize with blazing performance
const db = new JSONDatabase('data.json');

// Write data instantly
await db.set('user.1', {
  name: 'Alice',
  role: 'admin'
});

// Query with chainable API
const admins = await db.query('users')
  .where('role').eq('admin')
  .limit(10).exec();
